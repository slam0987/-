from datetime import datetime def calculate_age(birthdate): # تحويل الإدخال إلى تاريخ birthdate = datetime.strptime(birthdate, "%Y-%m-%d") # الحصول على التاريخ الحالي today = datetime.today() # حساب العمر age = today.year - birthdate.year - ((today.month, today.day) < (birthdate.month, birthdate.day)) return age # إدخال تاريخ الميلاد بصيغة YYYY-MM-DD birthdate = input("أدخل تاريخ ميلادك (YYYY-MM-DD): ") age = calculate_age(birthdate) print(f"عمرك هو: {age} سنة.")
